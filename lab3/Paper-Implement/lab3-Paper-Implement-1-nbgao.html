def PCA_whitening(X):
    # 0
    X -= np.mean(X)
    # 
    cov = np.dot(X.T, X) / X.shape[0]
    # SVD
    U,S,V = np.linalg.svd(cov)
    X_rot = np.dot(X, U) 
    X_white = X_rot / np.sqrt(S + 1e-6)
    return X

def zca_whitening(inputs):
    sigma = np.dot(inputs, inputs.T)/inputs.shape[1] #inputs
    U,S,V = np.linalg.svd(sigma) #
    epsilon = 0.1                #0
    ZCAMatrix = np.dot(np.dot(U, np.diag(1.0/np.sqrt(np.diag(S) + epsilon))), U.T)                     #zca
    return np.dot(ZCAMatrix, inputs)   #
